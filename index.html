Peaje de Aparcamiento Automático

Proyecto con Arduino UNO, sensores y automatización

Descripción del Proyecto

Este proyecto implementa un sistema de peaje de aparcamiento automático utilizando un sensor ultrasónico, un servomotor y un Arduino UNO para detectar vehículos y levantar la barrera sin intervención humana.

Código del Proyecto


#include <Servo.h>

Servo barrera;

int trig = 2;
int echo = 3;
long duracion;
int distancia;

void setup() {
  Serial.begin(9600);

  pinMode(trig, OUTPUT);
  pinMode(echo, INPUT);

  barrera.attach(4);
  barrera.write(90);
}

void loop() {

  digitalWrite(trig, LOW);
  delayMicroseconds(2);
  digitalWrite(trig, HIGH);
  delayMicroseconds(10);
  digitalWrite(trig, LOW);

  duracion = pulseIn(echo, HIGH, 30000);
  distancia = duracion * 0.0343 / 2;

  Serial.print("Distancia: ");
  Serial.println(distancia);

  if (distancia > 0 && distancia < 15) {
    abrirBarrera();
    delay(10000);
    cerrarBarrera();
  }

  delay(200);
}

void abrirBarrera() {
  barrera.write(0);
}

void cerrarBarrera() {
  barrera.write(90);
}
